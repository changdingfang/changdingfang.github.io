条件变量

  • std::condition_variable

详细一点的内容可以查看c++并发并行

使用示例

#include <stdlib.h>
#include <iostream>
#include <thread>
#include <queue>
#include <condition_variable>
#include <mutex>

int main(void)
{
    std::queue<int> nums;
    std::mutex mtx;
    std::condition_variable cv;
    bool notified = false;
    srandom(time(nullptr));

    /* 生产者 */
    auto producer = [&]() 
    {
        while (true)
        {
            std::this_thread::sleep_for(std::chrono::milliseconds(1000));
            std::unique_lock<std::mutex> lock(mtx);
            int num = rand() % 100;
            std::cout << "producing: " << num << std::endl;
            nums.push(num);
            notified = true;
            cv.notify_all();
        }
    };

    /* 消费者 */
    auto consumer = [&](int name) 
    {
        while (true)
        {
            std::unique_lock<std::mutex> lock(mtx);
            while (!notified)
            {
                cv.wait(lock);
            }
            lock.unlock();
            std::this_thread::sleep_for(std::chrono::milliseconds(2500));
            lock.lock();
            if (!nums.empty())
            {
                std::cout << "cosuming " << name << ": " << nums.front() << std::endl;
                nums.pop();
            }
            notified = false;
        }
    };

    const int consumer_num = 4;
    std::thread p(producer);
    std::thread cs[consumer_num];

    for (int i = 0; i < consumer_num; ++i)
    {
        cs[i] = std::thread(consumer, i);
    }

    p.join();
    for (int i = 0; i < consumer_num; ++i)
    {
        cs[i].join();
    }

    return 0;
}
