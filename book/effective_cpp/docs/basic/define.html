尽量以const,enum,inline替换#define

"#define"不被视为语言的一部分。 

#define PI 3.14

记号PI可能没有进入记号表（symbol table）内。 所以在编译错误信息时，可能会产生困惑，因为看到的信息也许是3.14而不是PI

使用常量替换

/* c++11之后出现的constexpr, 11之前使用const */
constexpr double PI = 3.14;

/* 常量字符串*/
const char *const name = "ding fang"; /* 或 const std::string("ding fang"); */

/* 一个class的专属常量 */
class A
{
private:
    static constexpr int num;  /* static 避免产生多份实体,需要在实现文件中定义 */
    /* 或者用枚举 */
    enum { Num = 99 };  /* 可以避免被获取地或被或被引用, 不会分配非必要的内存 */
};
constexpr int A::num = 99;

inline 替代 函数宏

#include <iostream>

#define MAX(a, b) ( ((a) > (b)) ? (a) : (b) )

template<typename T>
inline T Max(T &a, T &b)
{
    return a > b ? a : b;
}

int main(void)
{
    int a = 2, b = 0;
    /* 
     * a被加了两次,
     * 因为被替换为 (++a) > (b) ? (++a) : (b) 
     * 当a < b时a只被加一次
     * */
    std::cout << MAX(++a, b) << std::endl;
    std::cout << a << std::endl; 

    int c = 2, d = 0;
    /* 使用inline也可以达到替换代码的效果，但不会有define一样的诡异行为 */
    std::cout << Max(++c, d) << std::endl;
    std::cout << c << std::endl;

    return 0;
}
