    字段

    字段的增删查改

    去除主键

    alter table table_test drop primary key;

    添加字段

    alter table table1 add transactor varchar(10) not Null;
alter table table1 add id int unsigned not Null auto_increment primary key      // 添加字段并设为主键

    删除字段

    ALTER TABLE `table_name` DROP 字段名

    修改字段位置

    alter table `table_name` modify 字段名 int(10) unsigned auto_increment first;        // 到第一位
alter table `table_name` modify 字段名 varchar(10) after 某一位字段;           // 到某一位字段后面

    修改字段类型

    alter table 表名称 change 字段原名称 字段新名称 字段类型 [是否允许非空];
alter table 表名称 modify 字段名称 字段类型 [是否允许非空];
alter table 表名称 modify 字段名称 字段类型 [是否允许非空];

    表

    查看表结构

    desc `table_name`;

    复制表结构及数据到新表

    CREATE TABLE 新表 SELECT * FROM 旧表

    只复制表结构到新表

    CREATE TABLE 新表 SELECT * FROM 旧表 WHERE 1=2;      // 即: 让WHERE条件不成立.

    复制旧表的数据到新表

    INSERT INTO 新表 SELECT * FROM 旧表     // 旧表和新表结构一样
INSERT INTO 新表(字段1,字段2,.......) SELECT 字段1,字段2,...... FROM 旧表       // 旧表和新表结构不一样

    查看表状态

    show table status from db like 条件     // 可以查看engine数据库引擎, version, row, index等信息

    库

    查看和使用数据库

    show databases;             // 显示所有可见的数据库
use `database_name`;        // 切换到database_name数据库

    查看当前数据库的所有表

    show tables;
