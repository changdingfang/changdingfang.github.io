future

提供了访问异步操作结果的途径。

详细一点的内容可以查看c++并发并行

#include <unistd.h>
#include <iostream>
#include <thread>
#include <future>

int main(void)
{
    /* packaged_task也是c++11特性 */
    std::packaged_task<int()> task([]() { sleep(2); return 5; });
    std::future<int> f = task.get_future();
    std::thread(std::move(task)).detach();

    std::cout << "waiting ..." << std::endl;

    f.wait();
    std::cout << "f result : " << f.get() << std::endl;

    return 0;
}
